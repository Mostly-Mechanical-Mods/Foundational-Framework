FF_recipient_can_be_councillor_of = {
	$COURT_OWNER$ = { save_temporary_scope_as = court_owner }
	is_capable_adult = yes
	is_imprisoned = no
	NOT = { is_at_war_with = scope:court_owner }
}

FF_has_ambition_get_council_position = {
	custom_description = {
		text = FF_has_ambition_get_council_position
		any_in_global_list = {
			variable = FF_council_position_datamodel
			switch = {
				trigger = this
				flag:chancellor = {
					prev = {
						can_be_chancellor_trigger = { COURT_OWNER = scope:actor }
						FF_ambition_get_council_position_chancellor > 0
					}
				}
				flag:steward = {
					prev = {
						can_be_steward_trigger = { COURT_OWNER = scope:actor }
						FF_ambition_get_council_position_steward > 0
					}
				}
				flag:marshal = {
					prev = {
						can_be_marshal_trigger = { COURT_OWNER = scope:actor }
						FF_ambition_get_council_position_marshal > 0
					}
				}
				flag:spymaster = {
					prev = {
						can_be_spymaster_trigger = { COURT_OWNER = scope:actor }
						FF_ambition_get_council_position_spymaster > 0
					}
				}
			}
		}
	}
}
